#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <math.h>
#include <RTClib.h>       // DS3231 RTC

// ===================== Pins =====================
#define TRIG_PIN     5
#define ECHO_PIN     18
#define LED_PIN      2
#define BUZZER_PIN   15         // 3.3V active buzzer
#define RELAY_PIN    27         // Relay control (IN)

const bool RELAY_ACTIVE_LOW = false;   // your relay is active-HIGH

#define SDA_PIN 21
#define SCL_PIN 22

// ===================== OLED =====================
#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

// ===================== MPU6050 (manual I2C) ==================
const uint8_t MPU_ADDR = 0x69;   // confirmed by scanner
bool mpuOK = false;

#define MPU_REG_PWR_MGMT_1   0x6B
#define MPU_REG_ACCEL_XOUT_H 0x3B

// ===================== RTC DS3231 ===========================
RTC_DS3231 rtc;
bool rtcOK = false;
bool haveLastAlarmTime = false;
DateTime lastAlarmTime;

// Alarm cause tracking
const uint8_t CAUSE_NONE  = 0;
const uint8_t CAUSE_DIST  = 1;
const uint8_t CAUSE_SHAKE = 2;
uint8_t lastAlarmCause = CAUSE_NONE;

// Track previous “any alarm” state
bool prevAnyAlarm = false;

// ===================== Thresholds =================
const float DIST_ON_CM     = 10.0;   // distance alarm when > 10cm
const float SHAKE_THRESH_G = 0.30;   // shake alarm when >= 0.30 g

// ===================== State =====================
bool lockLatched      = false;  // once true, lock stays on
bool prevShakeAlarm   = false;  // for edge detection
bool inShakeBeep      = false;  // true while running 10-beep sequence

// ===================== Buzzer Pulse for DISTANCE alarm =================
const uint16_t BUZZ_ON_MS  = 150;
const uint16_t BUZZ_OFF_MS = 200;
bool buzzerState = false;
unsigned long buzzTimer = 0;

// ===================================================
void setup() {
  Serial.begin(115200);

  pinMode(TRIG_PIN, OUTPUT);
  pinMode(ECHO_PIN, INPUT);
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(RELAY_PIN, OUTPUT);

  digitalWrite(LED_PIN, LOW);
  digitalWrite(BUZZER_PIN, LOW);
  setRelay(false);

  Wire.begin(SDA_PIN, SCL_PIN);

  // ---------- OLED ----------
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("SSD1306 init failed!");
    while (1) delay(10);
  }
  display.clearDisplay();
  display.setTextColor(SSD1306_WHITE);
  display.setTextSize(1);

  // ---------- RTC DS3231 ----------
  if (!rtc.begin()) {
    Serial.println("RTC DS3231 not found!");
    rtcOK = false;
  } else {
    rtcOK = true;
    Serial.println("RTC DS3231 OK at 0x68");

    if (rtc.lostPower()) {
      Serial.println("RTC lost power, setting time to compile time.");
      rtc.adjust(DateTime(F(__DATE__), F(__TIME__)));
    }
  }

  // ---------- MPU6050 ----------
  Serial.println("Checking MPU6050 at 0x69...");
  Wire.beginTransmission(MPU_ADDR);
  uint8_t err = Wire.endTransmission();
  if (err == 0) {
    Wire.beginTransmission(MPU_ADDR);
    Wire.write(MPU_REG_PWR_MGMT_1);
    Wire.write(0x00);  // wake up
    Wire.endTransmission();
    mpuOK = true;
    Serial.println("MPU6050 ACK at 0x69, init done.");
  } else {
    mpuOK = false;
    Serial.println("MPU6050 did not ACK, running without shake detection.");
  }

  Serial.println("Setup complete.");
}

// ===================== Helpers =====================
void setRelay(bool on) {
  bool level = RELAY_ACTIVE_LOW ? !on : on;
  digitalWrite(RELAY_PIN, level);
}

float readDistanceCm() {
  digitalWrite(TRIG_PIN, LOW);
  delayMicroseconds(2);
  digitalWrite(TRIG_PIN, HIGH);
  delayMicroseconds(10);
  digitalWrite(TRIG_PIN, LOW);

  long duration = pulseIn(ECHO_PIN, HIGH, 30000); // 30ms timeout (~5m)
  if (duration == 0) return NAN;
  return (duration * 0.034f) / 2.0f;              // cm
}

// read shake magnitude in g
float readShakeG() {
  if (!mpuOK) return NAN;

  Wire.beginTransmission(MPU_ADDR);
  Wire.write(MPU_REG_ACCEL_XOUT_H);
  if (Wire.endTransmission(false) != 0) return NAN;

  Wire.requestFrom((int)MPU_ADDR, 6);
  if (Wire.available() < 6) return NAN;

  int16_t ax = (Wire.read() << 8) | Wire.read();
  int16_t ay = (Wire.read() << 8) | Wire.read();
  int16_t az = (Wire.read() << 8) | Wire.read();

  const float LSB_PER_G = 16384.0f;  // ±2g range

  float ax_g = ax / LSB_PER_G;
  float ay_g = ay / LSB_PER_G;
  float az_g = az / LSB_PER_G;

  float mag_g = sqrtf(ax_g*ax_g + ay_g*ay_g + az_g*az_g);
  float shake_g = fabsf(mag_g - 1.0f); // remove gravity

  return shake_g;
}

// compute alarms for this cycle
void computeAlarm(float distance, float shake,
                  bool &distAlarmNow, bool &shakeAlarmNow) {

  distAlarmNow  = (!isnan(distance) && distance > DIST_ON_CM);
  bool nearOrUnknown = (isnan(distance) || distance <= DIST_ON_CM);
  shakeAlarmNow = (nearOrUnknown && !isnan(shake) && shake >= SHAKE_THRESH_G);
}

// non-blocking pulsing for DISTANCE alarm
void buzzerPulseUpdate(bool distAlarmActive) {
  if (!distAlarmActive) {
    buzzerState = false;
    digitalWrite(BUZZER_PIN, LOW);
    return;
  }

  unsigned long now = millis();
  if (!buzzerState && now - buzzTimer >= BUZZ_OFF_MS) {
    buzzerState = true;
    buzzTimer = now;
    digitalWrite(BUZZER_PIN, HIGH);
  } else if (buzzerState && now - buzzTimer >= BUZZ_ON_MS) {
    buzzerState = false;
    buzzTimer = now;
    digitalWrite(BUZZER_PIN, LOW);
  }
}

// 10-beep alarm for SHAKE, lock latches immediately
void triggerShakeAlarm(float distance, float shake) {

  inShakeBeep = true;    // so distance alarm doesn’t fight for buzzer/LED

  Serial.println("=== SHAKE ALARM TRIGGERED ===");

  // latch lock immediately
  lockLatched = true;
  setRelay(true);        // engage relay RIGHT NOW

  // LED ON during beeps
  digitalWrite(LED_PIN, HIGH);

  const int NUM_BEEPS   = 10;
  const int BEEP_ON_MS  = 150;
  const int BEEP_OFF_MS = 150;

  for (int i = 0; i < NUM_BEEPS; i++) {
    digitalWrite(BUZZER_PIN, HIGH);
    delay(BEEP_ON_MS);
    digitalWrite(BUZZER_PIN, LOW);
    delay(BEEP_OFF_MS);
  }

  // After 10 beeps: LED OFF, buzzer already off.
  digitalWrite(LED_PIN, LOW);

  inShakeBeep = false;
}

// OLED status display
void showOnOLED(float dist, float shake,
                bool distAlarm, bool shakeAlarm) {

  display.clearDisplay();

  // Line 1: distance
  display.setCursor(0, 0);
  display.print("Dist: ");
  if (isnan(dist)) display.print("--");
  else { display.print(dist, 1); display.print(" cm"); }

  // Line 2: shake
  display.setCursor(0, 12);
  display.print("Shake: ");
  if (!mpuOK || isnan(shake)) display.print("N/A");
  else { display.print(shake, 2); display.print(" g"); }

  // Line 3: distance alarm status
  display.setCursor(0, 24);
  display.print("DistAlarm: ");
  display.print(distAlarm ? "YES" : "NO");

  // Line 4: lock state
  display.setCursor(0, 36);
  display.print("Lock: ");
  display.print(lockLatched ? "ON" : "OFF");

  // Line 5: last alarm time + cause
  display.setCursor(0, 48);
  display.print("Last ");

  if (rtcOK && haveLastAlarmTime) {
    // Cause: S = Shake, D = Distance
    if (lastAlarmCause == CAUSE_SHAKE)      display.print("S ");
    else if (lastAlarmCause == CAUSE_DIST)  display.print("D ");
    else                                    display.print("? ");

    // Time HH:MM:SS
    if (lastAlarmTime.hour() < 10) display.print('0');
    display.print(lastAlarmTime.hour());
    display.print(':');
    if (lastAlarmTime.minute() < 10) display.print('0');
    display.print(lastAlarmTime.minute());
    display.print(':');
    if (lastAlarmTime.second() < 10) display.print('0');
    display.print(lastAlarmTime.second());
  } else {
    display.print("-- --:--:--");
  }

  display.display();
}

// ===================== Main Loop =====================
void loop() {
  float distance = readDistanceCm();
  float shake    = readShakeG();

  bool distAlarmNow, shakeAlarmNow;
  computeAlarm(distance, shake, distAlarmNow, shakeAlarmNow);

  // --- SHAKE ALARM: edge-triggered, latching lock ---
  if (shakeAlarmNow && !prevShakeAlarm) {
    triggerShakeAlarm(distance, shake);
  }
  prevShakeAlarm = shakeAlarmNow;

  // === ANY ALARM (distance OR shake) -> timestamp once per event ===
  bool anyAlarmNow = distAlarmNow || shakeAlarmNow;

  if (anyAlarmNow && !prevAnyAlarm && rtcOK) {
    // rising edge: an alarm just started
    lastAlarmTime = rtc.now();
    haveLastAlarmTime = true;
    lastAlarmCause = shakeAlarmNow ? CAUSE_SHAKE : CAUSE_DIST;

    Serial.print("Alarm ");
    Serial.print((lastAlarmCause == CAUSE_SHAKE) ? "SHAKE" : "DIST");
    Serial.print(" at: ");
    Serial.print(lastAlarmTime.timestamp(DateTime::TIMESTAMP_DATE));
    Serial.print(" ");
    Serial.println(lastAlarmTime.timestamp(DateTime::TIMESTAMP_TIME));
  }
  prevAnyAlarm = anyAlarmNow;

  // --- DISTANCE ALARM: continuous, pulsing buzzer, LED on ---
  if (!inShakeBeep) {
    digitalWrite(LED_PIN, distAlarmNow ? HIGH : LOW);
    buzzerPulseUpdate(distAlarmNow);
  }

  // Lock relay: ON if latched, otherwise OFF
  setRelay(lockLatched);

  // Serial debug
  Serial.print("Dist: ");
  if (isnan(distance)) Serial.print("NaN");
  else Serial.print(distance, 1);
  Serial.print(" cm | Shake: ");
  if (!mpuOK || isnan(shake)) Serial.print("N/A");
  else Serial.print(shake, 2);
  Serial.print(" g | distAlarm=");
  Serial.print(distAlarmNow ? "1" : "0");
  Serial.print(" | shakeAlarm=");
  Serial.print(shakeAlarmNow ? "1" : "0");
  Serial.print(" | lockLatched=");
  Serial.println(lockLatched ? "1" : "0");

  showOnOLED(distance, shake, distAlarmNow, shakeAlarmNow);

  delay(50);
}
